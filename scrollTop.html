            require("mmAnimate", function(avalon) {
                avalon.define("fx", function(vm) {
                    vm.array = avalon.range(0, 1000)
                    vm.effect = function() {
                        avalon(this).animate({scrollTop: "+=150"}, 1000)
                    }
                })
                avalon.scan()
            })
