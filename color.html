颜色的动画

例子 
       require("mmAnimate", function(avalon) {
                var getRandomColor = function() {
                    return  '#'+('00000'+(Math.random()*0x1000000<<0).toString(16)).slice(-6);
                }
                avalon.define("fx", function(vm) {
                    vm.effect = function() {
                        avalon(this).fx({background: getRandomColor()}, 1000)
                    }

                })
                avalon.scan()
            })
